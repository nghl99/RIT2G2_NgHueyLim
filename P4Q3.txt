INCLUDE Irvine32.inc
.DATA
WORDS BYTE "In english : You like english and espresso, excellent !",0
CHGWORDS BYTE "egg", 0
I BYTE 'I',0
.CODE
MAIN PROC
     MOV ESI, OFFSET WORDS
	 MOV EDI, OFFSET CHGWORDS
	 MOV EAX, OFFSET I
	 MOV AL, [EAX]
	 CALL WRITECHAR
	 MOV ECX, LENGTHOF WORDS
L1:
     MOV AL, ' '
     CMP [ESI], AL
     JE L2
	 JMP SKIP
L2:
     MOV AL, 'e'
	 CMP [ESI+1], AL
	 JE CHANGE1
	 JMP SKIP
CHANGE1:
     MOV AL, [EDI]
     CALL WRITECHAR
	 JMP CHANGE2
CHANGE2:
     MOV AL, [EDI+1]
     CALL WRITECHAR
	 JMP CHANGE3
CHANGE3:
     MOV AL, [EDI+2]
     CALL WRITECHAR
	 JMP COMMA
SKIP:
    MOV AL, [ESI+1]
	CALL WRITECHAR
	JMP COMMA
COMMA:
     CMP ECX, 1
	 JE TERMINATE 
	 INC ESI
	 LOOP L1
TERMINATE:
     EXIT
MAIN ENDP

END MAIN
